#include <stdio.h>
 
int main() {
 
    int T = 0; 
    long long int fib[60];
    int i;
    int X = 0;
    int n1 = 0;
    int n2 = 1;
    
    fib[0] = 0;
    fib[1] = 1;
    
    for(int i = 2; i<= 60; i++)
    {
        fib[i] = fib[i-1] + fib[i-2];
    }
    
    scanf ("%d", &T);
    
    for(i = 1; i<=T; i++)
    {
        scanf("%d",&X);
        printf ("Fib(%d) = %lld\n", X, fib[X]);   
    }

 
    return 0;
}
