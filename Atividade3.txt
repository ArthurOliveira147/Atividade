 1005:Média 1

    #include <stdio.h>

  #define PESO_1 3.5
  #define PESO_2 7.5
  
int main() {
 
   double a = 0.0, b = 0.0;
   double media = 0.0;
   
   scanf("%lf\n%lf", &a, &b);
   media = ((a*PESO_1)+(b*PESO_2))/(PESO_1+PESO_2);
   printf("MEDIA = %.5lf\n",media);
 
    return 0; 
}

 1006:Média 2

  #include <stdio.h>

  #define PESO_A 2
  #define PESO_B 3
  #define PESO_C 5
  
int main() {
 
   double a = 0.0, b = 0.0, c = 0.0;
   double media = 0.0;
   
   scanf("%lf\n%lf\n%lf", &a, &b, &c);
   media = ((a*PESO_A)+(b*PESO_B)+(c*PESO_C))/(PESO_A+PESO_B+PESO_C);
   printf("MEDIA = %.1lf\n",media);
 
    return 0; 
}
